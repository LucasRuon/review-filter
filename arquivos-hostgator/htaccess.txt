# Opina Já - Configurações Apache

# Habilitar Rewrite Engine
RewriteEngine On

# Redirecionar /privacy para privacy.html
RewriteRule ^privacy$ privacy.html [L]

# Redirecionar /terms para terms.html
RewriteRule ^terms$ terms.html [L]

# Redirecionar www para não-www (opcional, mas recomendado)
RewriteCond %{HTTP_HOST} ^www\.opinaja\.com\.br$ [NC]
RewriteRule ^(.*)$ https://opinaja.com.br/$1 [R=301,L]

# Forçar HTTPS (opcional, mas recomendado para segurança)
RewriteCond %{HTTPS} off
RewriteRule ^(.*)$ https://%{HTTP_HOST}%{REQUEST_URI} [L,R=301]
